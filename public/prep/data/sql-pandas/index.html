    import pandas as pd

df = pd.read_csv("./z-001.csv")
df.columns = df.columns.str.replace(" ", "")  # カラム名に含まれるスペースは消す
df.head()

    SQLで操作できる形式にする #

    pandas.DataFrame.to_sql

    from sqlite3 import connect

conn = connect(":memory:")
df.to_sql("Table1", conn)

    pandas.read_sql

    pd.read_sql("SELECT 平成21年度 FROM Table1", conn)
